                如果数组里是对象
                let aaa = [{name: 'aaa', id: 1},......]
                // indexOf获取index
                function getIndex (_arr,_obj) {
                    for(var i = 0; i < _arr.length; i++)
                    {
                        if(_arr[i] === _obj) {
                            return parseInt(i);
                        }
                    }
                    return -1;
                };

                let index = getIndex(aaa, obj)

                if (index !== -1) {
                  aaa.splice(index, 1)
                }
